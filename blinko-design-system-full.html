ConfigProvider 全局配置

ConfigProvider 提供全局的组件前缀、语言等配置。

  componentSize="default"
  locale="zh-CN"
>
  <App />
</ConfigProvider>

/* 可配置项 */
interface ConfigProviderProps {
  prefixCls?: string;          // 组件类名前缀
  componentSize?: string;     // 组件大小
  locale?: string;            // 语言
  autoInsertSpaceInButton?: boolean;  // 自动插入空格
  focusing?: boolean;         // 是否自动聚焦
  empty?: {                  // 空状态配置
    description?: ReactNode;
    img?: string;
    simple?: boolean;
  };
}
